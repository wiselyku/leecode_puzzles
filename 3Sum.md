# 15. 3Sum

https://leetcode.com/problems/3sum/

Basic Idea: Use the idea of Two sum question to solve this question.  My implemetation is not good enough because it is only faster than 5% people.

Method 1:
```python=
class Solution:

    def twoSum(self, target: int, hash_table) -> List[List[int]]:
        
        answer = []
        pair = []
               
        for key in hash_table:
                tem = target - key
                if tem in hash_table:
                    if tem == key and hash_table[tem]>=2:
                        pair = [tem, tem]
                        answer.append(pair)
                    elif tem != key:
                        pair = [tem, key]
                        pair.sort()
                        if pair not in answer:
                            answer.append(pair)
        
        return answer
                        
    
    def threeSum(self, nums: List[int]) -> List[List[int]]:
        hash_table = dict()
        for i in range(len(nums)):
            if nums[i] in hash_table:
                hash_table[nums[i]] += 1
            else:
                hash_table[nums[i]] = 1
        
        temp = []
        temp_answer = []
        answer = []
        for key in hash_table:
            target = 0-key
            temp = self.twoSum(target, hash_table)
            for j in range(len(temp)):
                if key == temp[j][0] and key == temp[j][1] and hash_table[key]>=3:
                    temp_answer = [key, key, key]
                    if temp_answer not in answer:
                        answer.append(temp_answer)
                        
                elif key == temp[j][0] and key != temp[j][1] and hash_table[key]>=2:
                    temp_answer = [key, key, temp[j][1]]
                    temp_answer.sort()
                    if temp_answer not in answer:
                        answer.append(temp_answer)    
                
                elif key != temp[j][0] and key == temp[j][1] and hash_table[key]>=2:
                    temp_answer = [key, key, temp[j][0]]
                    temp_answer.sort()
                    if temp_answer not in answer:
                        answer.append(temp_answer)     
                        
                elif key != temp[j][0] and key != temp[j][1]:
                    temp_answer = [key, temp[j][0], temp[j][1]]
                    temp_answer.sort()
                    if temp_answer not in answer:
                        answer.append(temp_answer)                      
                    
                
        return answer
```

```python=
class Solution:
    def threeSum(self, nums: List[int]) -> List[List[int]]:
        nums.sort()
        triplets = []
        for i in range(len(nums)):
            if i > 0 and nums[i] == nums[i - 1]:
                continue
            Solution.search_pair(nums, -nums[i], i + 1, triplets)
        return triplets
    
    def search_pair(nums, target_sum, left, return_array):
        right = len(nums) - 1
        while (left < right):
            current_sum = nums[left] + nums[right]
            if current_sum == target_sum:
                return_array.append([-target_sum, nums[left], nums[right]])
                left += 1
                right -= 1
                while left < right and nums[left] == nums[left - 1]:
                    left += 1
                while left < right and nums[right] == nums[right + 1]:
                    right -= 1
            elif target_sum > current_sum:
                left += 1
            else:
                right -= 1
```
